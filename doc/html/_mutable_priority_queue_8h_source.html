90 template <class T>
91 void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
92  auto x = H[i];
93  while (true) {
94  unsigned k = leftChild(i);
95  if (k >= H.size())
96  break;
97  if (k + 1 < H.size() && *H[k + 1] < *H[k])
98  k++; // right child of i
99  if (!(*H[k] < *x))
100  break;
101  set(i, H[k]);
102  i = k;
103  }
104  set(i, x);
105 }
